def isprime(x):
    if x <= 1:
        return False
    else:
        for i in range(2,x):
            if x%i == 0:
                return False
    return True
try:
    a = int(input("a: "))
    b = int(input("b: "))
except ValueError:
    print("Hibás bevitel!")

if a < b:
    oszto = a
else:
    oszto = b
ez = 0
for i in range(1,oszto+1):
    if a%i==0 and b%i==0 and isprime(i):
        ez = i

if ez == 0:
    print("Ennek a két számnak nincs közös prím osztója.")
else:
    print("A két szám legnagyobb közös prím osztója: {}".format(ez))
